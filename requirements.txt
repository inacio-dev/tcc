# ============================================================================
# DEPENDÊNCIAS DO CLIENT (codigo/client/)
# ============================================================================
# IMPORTANTE: Tkinter não pode ser instalado via pip no Linux!
# Execute ANTES de instalar via pip:
#   sudo apt-get install python3-tk
#
# Ou no Arch Linux:
#   sudo pacman -S tk
# ============================================================================

# OpenCV para processamento de vídeo e exibição
opencv-python>=4.8.0

# NumPy - versão compatível com simplejpeg do sistema (1.24.2)
# IMPORTANTE: Não instalar numpy mais novo que pode causar incompatibilidade
numpy==1.24.2

# Pillow para processamento de imagem otimizado (Tkinter)
Pillow>=10.0.0

# PySerial para comunicação serial com ESP32 (cockpit)
pyserial>=3.5

# ============================================================================
# DEPENDÊNCIAS DO RASPBERRY PI (codigo/raspberry/)
# ============================================================================
# NOTA: Estas dependências são apenas para o Raspberry Pi
# Não instale no client se não for necessário
# ============================================================================

# Camera do Raspberry Pi - usar versão do sistema
picamera2

# PCA9685 PWM Driver para controle de servos
adafruit-circuitpython-pca9685

# Dependências do CircuitPython para Raspberry Pi
adafruit-blinka

# Dependências do CircuitPython para servos
adafruit-circuitpython-motor
adafruit-circuitpython-servokit

# Bibliotecas I2C para BMI160 - usar versão do sistema
smbus2

# ============================================================================
# INSTALAÇÃO
# ============================================================================
# CLIENT (Linux):
#   1. sudo apt-get install python3-tk  # Instalar Tkinter
#   2. pip install opencv-python numpy==1.24.2 Pillow pyserial
#
# RASPBERRY PI:
#   1. sudo apt-get install python3-tk python3-opencv python3-numpy python3-picamera2
#   2. pip install -r requirements.txt
# ============================================================================
